// components/Logo.tsx
interface LogoProps {
  size?: number;
  theme?: 'light' | 'dark';
  className?: string;
}

export function Logo({ size = 32, theme = 'light', className }: LogoProps) {
  const bg = theme === 'light' ? '#E8580C' : '#F0874A';
  const fg = theme === 'light' ? '#FFFFFF' : '#0F0F17';

  // Stroke width mais grosso em tamanhos menores
  const sw = size <= 20 ? 4 : size <= 32 ? 3.5 : 3;
  const cr = size <= 20 ? 5 : size <= 32 ? 4.5 : 4;

  return (
    <svg
      width={size} height={size}
      viewBox="0 0 42 42" fill="none"
      className={className}
    >
      <rect width="42" height="42" rx="10" fill={bg} />
      <path
        d="M13 29A11.5 11.5 0 0 1 21 9.5"
        stroke={fg} strokeOpacity={0.35}
        strokeWidth={sw} strokeLinecap="round"
      />
      <path
        d="M21 9.5A11.5 11.5 0 0 1 32.5 21"
        stroke={fg} strokeOpacity={0.6}
        strokeWidth={sw} strokeLinecap="round"
      />
      <path
        d="M32.5 21A11.5 11.5 0 0 1 29 29"
        stroke={fg} strokeWidth={sw}
        strokeLinecap="round"
      />
      <circle cx="21" cy="22" r={cr} fill={fg} />
    </svg>
  );
}

Theme Toggle — implementação

// hooks/useTheme.ts
import { useState, useEffect } from 'react';

export function useTheme() {
  const [theme, setTheme] = useState<'light' | 'dark'>('light');

  useEffect(() => {
    // Checar localStorage primeiro, depois preferência do OS
    const saved = localStorage.getItem('widia-theme');
    if (saved) {
      setTheme(saved as 'light' | 'dark');
    } else if (window.matchMedia('(prefers-color-scheme: dark)').matches) {
      setTheme('dark');
    }
  }, []);

  useEffect(() => {
    document.documentElement.classList.toggle('dark', theme === 'dark');
    localStorage.setItem('widia-theme', theme);
  }, [theme]);

  const toggle = () => setTheme(t => t === 'light' ? 'dark' : 'light');

  return { theme, toggle };
}
07 — Resumo
Checklist de Implementação
## O que implementar:

✅ Design Tokens    → globals.css com :root e .dark
✅ Logo Arc Score    → Componente Logo.tsx (light/dark, tamanhos)
✅ Fontes            → Outfit (display) + DM Sans (body) + JetBrains Mono (números)
✅ Ícones            → lucide-react, stroke-width 1.8
✅ Áreas da Vida     → 6 cores semânticas com variantes light/dark
✅ Camadas           → bg → surface → card (sempre com border)
✅ Theme Toggle      → .dark no html + localStorage + prefers-color-scheme
✅ Border Radius     → 8 / 12 / 16 / 24px padronizado
✅ Sombras           → Light usa sombras, Dark usa bordas
✅ Scores/números    → font-mono (JetBrains Mono)

## Nunca fazer:

❌ Hardcode de cores — sempre usar var(--)
❌ Preto puro (#000) no dark mode
❌ Sombras pesadas no dark mode
❌ Misturar fontes além das 3 definidas
❌ Ícones sem strokeWidth={1.8}
❌ Cards sem border no dark mode
❌ Pular camadas de hierarquia visual
